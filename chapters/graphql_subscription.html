GraphQL Subscriptions


        #GraphQL subscription
        subscription PostSubscription($input: PostSubscriptionInput) {
            postSubscribe(input: $input) {
                comment {
                    author,
                    content
                }
            }
        }
    

        #GraphQL subscription result
        {
            "data": {
                "postSubscribe": {
                    "comment": {
                        "author": "foo",
                        "content": "bar"
                    }
                }
            }
        }
